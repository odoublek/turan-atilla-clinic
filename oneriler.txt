    <script>
      document.addEventListener("DOMContentLoaded", () => {
        const stepIndicators = document.querySelectorAll(".step-indicator");
        const progressBar = document.querySelector(".bar");
        const topSection = document.querySelector(".top-section");
        const stepImage = topSection.querySelector(".step-image img");
        const stepDescription = topSection.querySelector(".step-description");
        const totalSteps = stepIndicators.length;

        const stepData = [
          {
            title: "Consultation",
            description:
              "Consultations are always free. This is our opportunity to get to know you and your vision. And then share with you our expertise and recommendations based on 30+ years of experience, and thousands of satisfied patients.",
            comment:
              "My overall experience has been nothing less than beyond exceptional! [The PSC] made me feel confident and comfortable from start to finish…",
            image: "images/plastic-surgery-woman.jpg",
          },
          // Repeat for other steps...
        ];

        stepIndicators.forEach((indicator, index) => {
          indicator.addEventListener("click", () => {
            stepIndicators.forEach((indicator) =>
              indicator.classList.remove("active")
            );
            indicator.classList.add("active");

            const stepWidth = (100 / totalSteps) * (index + 1);
            progressBar.style.width = `${stepWidth}%`;

            // Update step image and title
            stepImage.src = stepData[index].image;
            stepDescription.innerHTML = `
        <h3 class="text-xl font-semibold mb-2">Step ${index + 1}: ${
              stepData[index].title
            }</h3>
        <p class="text-gray-600 mb-4">${stepData[index].description}</p>
        <div class="blockquote">
          <blockquote class="text-gray-600 border-l-4 border-teal-400 pl-4">
            Comment from one of our patients:<br />
            <em>${stepData[index].comment}</em>
          </blockquote>
        </div>
      `;

            // Scroll to the top of the step description
            topSection.scrollIntoView({ behavior: "smooth" });
          });
        });
      });
    </script>
